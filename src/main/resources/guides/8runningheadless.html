Installing Fiji

  • Linux
  • Mac
  • Windows 
# Commands for Bash
# Downloading the latest copy of Fiji for 64-bit Linux 
wget https://downloads.imagej.net/fiji/latest/fiji-linux64.zip

# Unzipping archive
unzip fiji-linux64.zip

# Removing downloaded zip archive
rm fiji-linux64.zip

# Moving into extracted directory
cd Fiji.app/
# Commands for Bash
# Downloading the latest copy of Fiji for 64-bit MacOS
curl -O https://downloads.imagej.net/fiji/latest/fiji-macosx.zip

# Unzipping archive
unzip fiji-macosx.zip -d 

# Removing downloaded zip archive
rm fiji-macosx.zip

# Moving into extracted directory
cd Fiji.app/Contents/MacOS
# Commands for Windows PowerShell
# Downloading the latest copy of Fiji for 64-bit Windows
Invoke-WebRequest https://downloads.imagej.net/fiji/latest/fiji-win64.zip

# Unzipping archive
Expand-Archive .\fiji-win64.zip

# Removing downloaded zip archive
Remove-Item .\fiji-win64.zip

# Moving into extracted directory
cd .\fiji-win64\Fiji.app

Installing MIA

  • Linux
  • Mac
  • Windows 
# Commands for Bash
# Marking MIA and IJPB-plugins for installation
./ImageJ-linux64 --update add-update-site MIA https://sites.imagej.net/ModularImageAnalysis/
./ImageJ-linux64 --update add-update-site IJPB-plugins https://sites.imagej.net/IJPB-plugins

# Performing update
./ImageJ-linux64 --update update
# Commands for Bash
# Marking MIA and IJPB-plugins for installation
./ImageJ-macosx --update add-update-site MIA https://sites.imagej.net/ModularImageAnalysis/
./ImageJ-macosx --update add-update-site IJPB-plugins https://sites.imagej.net/IJPB-plugins

# Performing update
./ImageJ-macosx --update update
# Commands for Windows PowerShell
# Marking MIA and IJPB-plugins for installation
.\ImageJ-win64 --console --update add-update-site MIA https://sites.imagej.net/ModularImageAnalysis/
.\ImageJ-win64 --console --update add-update-site IJPB-plugins https://sites.imagej.net/IJPB-plugins

# Performing update
.\ImageJ-win64 --console --update update
